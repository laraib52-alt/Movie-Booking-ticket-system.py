base_price = 15
age = int(input("enter your age:"))
seat_type =  input("enter your seat-type:")
show_time = input("enter your  show-time:")
is_member=input("you are a member(yes/no)?").lower() == "yes"
show_day=input("enter your show day:")
if age >= 18:
    print('User is eligible to book a ticket')
if age >= 21:
    print('User is eligible for Evening shows')
else:
    print('User is not eligible for Evening shows')
    print("ask somEonE adult to book tickEt")
discount = 0
if is_member and age >= 21:
    discount = 3
    print('User qualifies for membership discount')
else:
    print('User does not qualify for membership discount')
print('Discount:', discount)
is_weekend = show_day.lower() in ["saturday", "sunday"]

extra_charges = 0

if is_weekend or show_time == "Evening":
    extra_charges = 2
    print("Extra charges will be applied")
else:
    print("no extra charges")
extra_charges = 0
print('Extra charges:', extra_charges)

if age >= 18 and (show_time != "Evening" or age >= 21 or is_member):
    print("Ticket booking condition satisfied")

    service_charges = 0
    if seat_type == 'Premium':
        service_charges = 5
    elif seat_type == 'Gold':
        service_charges = 3
    else:
        service_charges = 1
    print('Service charges:', service_charges)
    final_price = (extra_charges + service_charges +base_price - discount)  
    print('Final price :', final_price)
else:
    print('Ticket booking failed due to restrictions')

print("Bye, Have a nice time")
